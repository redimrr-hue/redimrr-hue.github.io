let snake; let food; let scl = 20; function setup() { createCanvas(600, 400); frameRate(10); snake = new Snake(); placeFood(); } function draw() { background(30); if (snake.eat(food)) { placeFood(); } snake.update(); snake.death(); snake.show(); fill(255, 0, 0); rect(food.x, food.y, scl, scl); } function placeFood() { let cols = floor(width / scl); let rows = floor(height / scl); food = createVector( floor(random(cols)) * scl, floor(random(rows)) * scl ); } function keyPressed() { if (keyCode === UP_ARROW && snake.ydir !== 1) snake.dir(0, -1); if (keyCode === DOWN_ARROW && snake.ydir !== -1) snake.dir(0, 1); if (keyCode === LEFT_ARROW && snake.xdir !== 1) snake.dir(-1, 0); if (keyCode === RIGHT_ARROW && snake.xdir !== -1) snake.dir(1, 0); } class Snake { constructor() { this.body = [createVector(200, 200)]; this.xdir = 1; this.ydir = 0; this.len = 1; } dir(x, y) { this.xdir = x; this.ydir = y; } eat(pos) { let d = dist(this.body[0].x, this.body[0].y, pos.x, pos.y); if (d < 1) { this.len++; return true; } return false; } death() { let head = this.body[0]; for (let i = 1; i < this.body.length; i++) { if (head.x === this.body[i].x && head.y === this.body[i].y) { this.len = 1; this.body = [createVector(200, 200)]; } } } update() { let head = this.body[0].copy(); head.x += this.xdir * scl; head.y += this.ydir * scl; // screen wrap head.x = (head.x + width) % width; head.y = (head.y + height) % height; this.body.unshift(head); while (this.body.length > this.len) { this.body.pop(); } } show() { fill(0, 255, 120); for (let part of this.body) { rect(part.x, part.y, scl, scl); } } }
